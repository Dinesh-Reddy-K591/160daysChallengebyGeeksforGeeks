class Solution {
    public ArrayList<Integer> findMajority(int[] arr) {
        int n=arr.length;
        ArrayList<Integer>res=new ArrayList<>();
        HashMap<Integer,Integer>map=new HashMap<>();
        int count=0;
        for(int num:arr){
            map.put(num,map.getOrDefault(num,0)+1);
        }
        for(int key:map.keySet()){
            if(map.get(key)>n/3){
                res.add(key);
            }
        }
        Collections.sort(res);
        return res;
    }
}
