class Solution {
    public int missingNumber(int[] arr) {
        HashSet<Integer>set=new HashSet<>();
        for(int num:arr){
            if(num>0){
                set.add(num);
            }
        }
        int n=arr.length;
        for(int i=1;i<=n+1;i++){
            if(!set.contains(i)){
                return i;
            }
        }
        return -1;
    }
}
